105template <typename Scheduler>
106class SGD : public OptimizerWithScheduler<Scheduler> {
107 public:
114 SGD(std::vector<Tensor> params, Scheduler& scheduler,
115 float momentum_beta = 0.0f)
116 : OptimizerWithScheduler<Scheduler>(std::move(params), scheduler),
117 momentum_beta_(momentum_beta),
118 momentum_(this->params_.size()) {}
119
123 void step() override {
124 const float lr = this->scheduler_->get();
125 ++this->step_count_;
126
127 for (size_t idx = 0; idx < this->params_.size(); ++idx) {
128 Tensor& param = this->params_[idx];
129 if (!Optimizer::valid_param(param)) continue;
130 float* data = param.data();
131 const float* grad = param.grad();
132 const size_t n = param.numel;
133 if (momentum_beta_ != 0.0f) {
134 Optimizer::ensure_state_size(momentum_[idx], n);
135 auto& momentum_vec = momentum_[idx];
136 for (size_t i = 0; i < n; ++i) {
137 momentum_vec[i] = momentum_beta_ * momentum_vec[i] + grad[i];
138 data[i] -= lr * momentum_vec[i];
139 }
140 } else {
141 for (size_t i = 0; i < n; ++i) {
142 data[i] -= lr * grad[i];
143 }
144 }
145 }
146 }

148 private:
149 float momentum_beta_;
150 std::vector<std::vector<float>> momentum_;

161template <typename Scheduler>
162class Adam : public OptimizerWithScheduler<Scheduler> {
163 public:
174 Adam(std::vector<Tensor> params, Scheduler& scheduler, float beta1 = 0.9f,
175 float beta2 = 0.999f, float weight_decay = 0.0f, bool amsgrad = false,
176 float epsilon = 1e-8f)
177 : OptimizerWithScheduler<Scheduler>(std::move(params), scheduler),
178 beta1_(beta1),
179 beta2_(beta2),
180 weight_decay_(weight_decay),
181 amsgrad_(amsgrad),
182 epsilon_(epsilon),
183 m1_(this->params_.size()),
184 m2_(this->params_.size()),
185 vhat_(amsgrad ? this->params_.size() : 0) {}
186
190 void step() override {
191 const float lr = this->scheduler_->get();
192 ++this->step_count_;
193 const float bc1 =
194 1.0f - std::pow(beta1_, static_cast<float>(this->step_count_));
195 const float bc2 =
196 1.0f - std::pow(beta2_, static_cast<float>(this->step_count_));
197
198 for (size_t idx = 0; idx < this->params_.size(); ++idx) {
199 Tensor& param = this->params_[idx];
200 if (!Optimizer::valid_param(param)) continue;
201 float* data = param.data();
202 const float* grad_ptr = param.grad();
203 const size_t n = param.numel;
204 Optimizer::ensure_state_size(m1_[idx], n);
205 Optimizer::ensure_state_size(m2_[idx], n);
206 if (amsgrad_) {
207 Optimizer::ensure_state_size(vhat_[idx], n);
208 }
209 auto& m1_vec = m1_[idx];
210 auto& m2_vec = m2_[idx];
211 std::vector<float>* vhat_vec = amsgrad_ ? &vhat_[idx] : nullptr;
212
213 for (size_t i = 0; i < n; ++i) {
214 float grad = grad_ptr[i];
215 if (weight_decay_ != 0.0f) {
216 grad += weight_decay_ * data[i];
217 }
218
219 m1_vec[i] = beta1_ * m1_vec[i] + (1.0f - beta1_) * grad;
220 m2_vec[i] = beta2_ * m2_vec[i] + (1.0f - beta2_) * grad * grad;
221
222 float m1_hat = m1_vec[i] / bc1;
223 float m2_term = m2_vec[i];
224 if (amsgrad_) {
225 (*vhat_vec)[i] = std::max((*vhat_vec)[i], m2_vec[i]);
226 m2_term = (*vhat_vec)[i];
227 }
228 float m2_hat = m2_term / bc2;
229 data[i] -= lr * m1_hat / (std::sqrt(m2_hat) + epsilon_);
230 }
231 }
232 }
233
234 private:
235 float beta1_;
236 float beta2_;
237 float weight_decay_;
238 bool amsgrad_;
239 float epsilon_;
240 std::vector<std::vector<float>> m1_;
241 std::vector<std::vector<float>> m2_;
242 std::vector<std::vector<float>> vhat_;
243};
